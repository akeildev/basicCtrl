   ```python
   import asyncio
   import time
   from dataclasses import dataclass
   from typing import Optional, Any
   import anyio
   import structlog
   from PIL import Image

   from cua_overlay.state.graph import UIElement, Bbox
   from cua_overlay.ax.walker import walk_subtree, WalkResult
   from cua_overlay.ax.rate_limit import TokenBucket


   @dataclass
   class L2Snapshot:
       ocr_text: str
       walker_nodes: int
       walker_truncated: bool
       captured_at: float


   class L2Medium:
       """Vision OCR + depth-limited AX subtree.

       Per VERIFY-06: max 3 levels, 50 children, 500 nodes (the walker enforces these).
       Per Pitfall P3: NEVER full recursive — always walks via walk_subtree.

       Phase 1 scope: scaffolding wired and unit-tested. The Calculator demo (Plan 09)
       MUST NOT trigger L2 — L0+L1 alone produce confidence >= 0.50.
       """
       def __init__(self, bucket: Optional[TokenBucket] = None) -> None:
           self._bucket = bucket or TokenBucket()
           self._log = structlog.get_logger()

       async def snapshot(self, target: UIElement, ax_element: Any = None) -> L2Snapshot:
           t = time.monotonic()
           async with anyio.create_task_group() as tg:
               results: dict = {"ocr_text": "", "walker_nodes": 0, "walker_truncated": False}

               async def _ocr():
                   results["ocr_text"] = await self._capture_ocr(target.bbox)

               async def _walk():
                   if ax_element is not None:
                       wr = await walk_subtree(
                           ax_element,
                           pid=target.pid,
                           bundle_id=target.bundle_id,
                           bucket=self._bucket,
                       )
                       results["walker_nodes"] = len(wr.nodes)
                       results["walker_truncated"] = wr.truncated
                   else:
                       # No element provided: degrade gracefully.
                       results["walker_nodes"] = 0
                       results["walker_truncated"] = False

               tg.start_soon(_ocr)
               tg.start_soon(_walk)

           return L2Snapshot(
               ocr_text=results["ocr_text"],
               walker_nodes=results["walker_nodes"],
               walker_truncated=results["walker_truncated"],
               captured_at=t,
           )

       async def run(
           self,
           target: UIElement,
           ax_element: Any,
           before: L2Snapshot,
           expected_text: Optional[str] = None,
       ) -> dict[str, float]:
           after = await self.snapshot(target, ax_element)
           signals: dict[str, float] = {}

           # OCR text change
           if before.ocr_text and after.ocr_text:
               signals["l2.ocr_text_changed"] = 0.0 if before.ocr_text.strip() == after.ocr_text.strip() else 1.0
           else:
               signals["l2.ocr_text_changed"] = 0.0

           # Expected text presence (if caller specified one)
           if expected_text:
               signals["l2.expected_text_present"] = 1.0 if expected_text in after.ocr_text else 0.0

           # Subtree node count change
           delta = abs(after.walker_nodes - before.walker_nodes)
           signals["l2.subtree_size_changed"] = min(1.0, delta / max(1, before.walker_nodes))

           # Truncated flag — confidence-reducer per VERIFY-06
           signals["l2.walker_truncated"] = 1.0 if (before.walker_truncated or after.walker_truncated) else 0.0

           self._log.info("l2.medium_run", signals=signals,
                          before_nodes=before.walker_nodes, after_nodes=after.walker_nodes)
           return signals

       async def _capture_ocr(self, bbox: Bbox) -> str:
           """Use ocrmac to OCR a 100x100 ROI around bbox.centroid."""
           def _sync():
               try:
                   from ocrmac import ocrmac
                   from cua_overlay.verifier.ensemble.l1_cheap import L1Cheap
                   # Reuse the CGImage capture helper from L1Cheap
                   capture = L1Cheap()
                   # Direct approach: re-implement small ROI capture (we call capture._cgimage_to_pil below to convert the CGImage to PIL).
                   from Quartz import (
                       CGRectMake,
                       CGWindowListCreateImage,
                       kCGWindowListOptionOnScreenOnly,
                       kCGNullWindowID,
                       kCGWindowImageDefault,
                   )
                   cx, cy = bbox.centroid
                   rect = CGRectMake(cx - 50, cy - 50, 100, 100)
                   cg_image = CGWindowListCreateImage(
                       rect,
                       kCGWindowListOptionOnScreenOnly,
                       kCGNullWindowID,
                       kCGWindowImageDefault,
                   )
                   if cg_image is None:
                       return ""
                   pil_img = capture._cgimage_to_pil(cg_image)
                   if pil_img is None:
                       return ""
                   annotations = ocrmac.OCR(pil_img).recognize()
                   return " ".join(a[0] for a in annotations) if annotations else ""
               except Exception:
                   return ""
           return await asyncio.to_thread(_sync)
   ```
