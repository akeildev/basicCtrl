3. **Create `basicctrl/profile/cache.py`** with:
   ```python
   import json
   import os
   from pathlib import Path
   from typing import Optional
   from basicctrl.profile.classifier import AppProfile

   _CACHE_DIR = Path.home() / ".cua" / "profiles"

   def _cache_path(bundle_id: str, base: Optional[Path] = None) -> Path:
       d = base or _CACHE_DIR
       return d / f"{bundle_id}.json"

   def save_cached_profile(profile: AppProfile, base: Optional[Path] = None) -> Path:
       """Atomic write per Pitfall 16 (avoid torn writes on crash)."""
       path = _cache_path(profile.bundle_id, base)
       path.parent.mkdir(parents=True, exist_ok=True)
       tmp = path.with_suffix(".json.tmp")
       tmp.write_text(profile.model_dump_json(indent=2))
       os.replace(tmp, path)
       return path

   def load_cached_profile(bundle_id: str, base: Optional[Path] = None) -> Optional[AppProfile]:
       path = _cache_path(bundle_id, base)
       if not path.exists():
           return None
       try:
           return AppProfile.model_validate_json(path.read_text())
       except Exception:
           return None  # corrupt cache file: re-probe

   def should_invalidate_cache(cached: AppProfile, current_bundle_version: Optional[str], current_bundle_build: Optional[str]) -> bool:
       """Per Pitfall 16: schema/version drift breaks reads. Re-probe if EITHER version OR build differs."""
       if cached.bundle_version != current_bundle_version:
           return True
       if cached.bundle_build != current_bundle_build:
           return True
       return False
   ```

2. **Create `basicctrl/profile/capability_probe.py`** with these async functions (each wraps a slow native call in `asyncio.to_thread` for non-blocking):
   ```python
   import asyncio
   import plistlib
   from pathlib import Path
   from typing import Optional

   async def probe_bundle_metadata(bundle_id: str) -> dict:
       """Returns {bundle_path, bundle_version, bundle_build, info_plist}. Sync (~5ms)."""
       def _sync():
           from AppKit import NSWorkspace
           url = NSWorkspace.sharedWorkspace().URLForApplicationWithBundleIdentifier_(bundle_id)
           if url is None:
               return {"bundle_path": None, "bundle_version": None, "bundle_build": None, "info_plist": {}}
           bundle_path = url.path()
           plist_path = Path(bundle_path) / "Contents" / "Info.plist"
           info = plistlib.loads(plist_path.read_bytes()) if plist_path.exists() else {}
           return {
               "bundle_path": bundle_path,
               "bundle_version": info.get("CFBundleShortVersionString"),
               "bundle_build": info.get("CFBundleVersion"),
               "info_plist": info,
           }
       return await asyncio.to_thread(_sync)

   async def probe_ax_rich(pid: int) -> bool:
       """AXChildren on app element returns >0 within 200ms."""
       def _sync():
           from HIServices import AXUIElementCreateApplication, AXUIElementCopyAttributeValue
           from HIServices import kAXChildrenAttribute
           app = AXUIElementCreateApplication(pid)
           err, children = AXUIElementCopyAttributeValue(app, kAXChildrenAttribute, None)
           return err == 0 and children is not None and len(children) > 0
       try:
           return await asyncio.wait_for(asyncio.to_thread(_sync), timeout=0.2)
       except (asyncio.TimeoutError, Exception):
           return False

   async def probe_ax_observer_works(pid: int) -> bool:
       """Subscribe to kAXFocusedUIElementChanged; if observer never fires within 500ms, return False (Pitfall 14)."""
       # Phase 1 implementation: best-effort. Use AXObserverCreate; subscribe; trigger a focus change via
       # AppleScript "tell application id <bundle> to activate"; wait for callback or timeout.
       # If subscribe itself errors with kAXErrorNotificationUnsupported, return False immediately.
       # NOTE: Plan 04 builds the full AXEventBridge; here we use a minimal local one-shot observer.
       def _sync():
           from HIServices import (
               AXUIElementCreateApplication, AXObserverCreate,
               AXObserverAddNotification, AXObserverGetRunLoopSource,
           )
           # Minimal probe — many web/Electron apps return kAXErrorNotificationUnsupported here.
           import threading
           result = {"fired": False, "supported": True}
           # ... full body in implementation; if any error path returns kAXErrorNotificationUnsupported
           # set supported=False and return False.
           return result
       try:
           res = await asyncio.wait_for(asyncio.to_thread(_sync), timeout=0.5)
           return res.get("fired", False) and res.get("supported", True)
       except (asyncio.TimeoutError, Exception):
           return False

   async def probe_cdp_ports(pid: int) -> Optional[int]:
       """Try connecting to localhost:9222..9230 with 100ms each. Return first responding port."""
       import httpx
       async with httpx.AsyncClient(timeout=0.1) as client:
           for port in range(9222, 9231):
               try:
                   r = await client.get(f"http://localhost:{port}/json/version")
                   if r.status_code == 200:
                       return port
               except Exception:
                   continue
       return None

   def probe_applescript_sdef(info_plist: dict) -> bool:
       return info_plist.get("OSAScriptingDefinition") is not None or info_plist.get("NSAppleScriptEnabled") is True

   def probe_electron(bundle_path: Optional[str]) -> bool:
       if not bundle_path:
           return False
       return (Path(bundle_path) / "Contents" / "Frameworks" / "Electron Framework.framework").exists()

   def probe_tauri_or_wails(bundle_path: Optional[str], info_plist: dict) -> bool:
       """[ASSUMED A2 from research] Heuristic: WebKit.framework linked + no .sdef + not Electron.
       Phase 1 spike. If wrong, mis-classifies Tauri as native AppKit; mitigation: Plan 02 logs a structlog
       warning when this heuristic fires so we can audit on real samples."""
       if not bundle_path:
           return False
       if probe_electron(bundle_path):
           return False
       if probe_applescript_sdef(info_plist):
           return False
       webkit_present = (Path(bundle_path) / "Contents" / "Frameworks" / "WebKit.framework").exists()
       return webkit_present
   ```

3. **Update `basicctrl/profile/classifier.py`** to add the real `classify()` function:
   ```python
   import time
   from datetime import datetime, timezone
   import anyio
   import structlog
   from basicctrl.profile.cache import load_cached_profile, save_cached_profile, should_invalidate_cache
   from basicctrl.profile.capability_probe import (
       probe_bundle_metadata, probe_ax_rich, probe_ax_observer_works,
       probe_cdp_ports, probe_applescript_sdef, probe_electron, probe_tauri_or_wails,
   )
   from basicctrl.profile.tcc import TCCMonitor

   _tcc = TCCMonitor()

   async def classify(bundle_id: str, pid: int) -> AppProfile:
       """Phase 1 main entry-point. <500ms first probe; <5ms cached."""
       t_start = time.monotonic()
       log = structlog.get_logger().bind(bundle_id=bundle_id, pid=pid)

       # T-1-04 mitigation: TCC check at every classify() call.
       if not await _tcc.check():
           await _tcc.on_revocation()  # raises SystemExit(2)

       # FAST PATH: bundle metadata first (~5ms sync), then check cache.
       meta = await probe_bundle_metadata(bundle_id)
       cached = load_cached_profile(bundle_id)
       if cached and not should_invalidate_cache(cached, meta["bundle_version"], meta["bundle_build"]):
           log.info("appprofile_cache_hit", elapsed_ms=int((time.monotonic() - t_start) * 1000))
           return cached

       # PARALLEL PROBES (anyio task group, each with 200ms timeout).
       results: dict = {}
       async with anyio.create_task_group() as tg:
           async def _ax():
               results["ax_rich"] = await probe_ax_rich(pid)
           async def _obs():
               results["ax_observer_works"] = await probe_ax_observer_works(pid)
           async def _cdp():
               results["cdp_port"] = await probe_cdp_ports(pid) if probe_electron(meta["bundle_path"]) else None
           tg.start_soon(_ax)
           tg.start_soon(_obs)
           tg.start_soon(_cdp)

       is_electron = probe_electron(meta["bundle_path"])
       has_sdef = probe_applescript_sdef(meta["info_plist"])
       is_tauri = probe_tauri_or_wails(meta["bundle_path"], meta["info_plist"])
       if is_tauri:
           log.warning("tauri_or_wails_heuristic_fired", bundle=bundle_id)

       # Derive translator priority.
       priority: list[str] = []
       if is_electron and results["cdp_port"] is not None:
           priority.append("T2")  # CDP best for Electron
       if results["ax_rich"]:
           priority.append("T1")  # AX
       if has_sdef:
           priority.append("T3")  # AppleScript
       priority.append("T4")  # Vision/Screen2AX always available
       priority.append("T5")  # Pixel always available

       profile = AppProfile(
           bundle_id=bundle_id,
           bundle_version=meta["bundle_version"],
           bundle_build=meta["bundle_build"],
           bundle_path=meta["bundle_path"],
           ax_rich=results["ax_rich"],
           ax_observer_works=results["ax_observer_works"],
           applescript_sdef=has_sdef,
           cdp_port=results["cdp_port"],
           cdp_available_after_relaunch=is_electron and results["cdp_port"] is None,
           tauri_or_wails=is_tauri,
           electron=is_electron,
           tcc_axenabled=True,  # we already checked above; if it had been False we'd have exited
           translator_priority=priority,
           probed_at=datetime.now(timezone.utc),
           probe_latency_ms=int((time.monotonic() - t_start) * 1000),
       )
       save_cached_profile(profile)
       log.info("appprofile_probed", priority=priority, latency_ms=profile.probe_latency_ms)
       return profile
   ```
