   ```python
   import asyncio
   import threading
   import time
   from dataclasses import dataclass, field
   from typing import Any, Callable, Optional
   import structlog

   from cua_overlay.ax.errors import AXError, axerror_from_code

   @dataclass(frozen=True)
   class AXEvent:
       pid: int
       element_key: str
       notif: str
       user_info: Any
       event_ts_ns: int
       action_id: Optional[str]

   @dataclass
   class Subscription:
       pid: int
       element_key: str
       notifications: list[str]
       action_id: str
       subscription_ts_ns: int     # Pitfall P28 anchor
       _observer: Any = None       # AXObserver opaque
       _runloop_source: Any = None
       _filter: Optional[Callable[[AXEvent], bool]] = None

   class AXEventBridge:
       """CFRunLoop thread + asyncio Queue bridge."""
       def __init__(self, loop: asyncio.AbstractEventLoop) -> None:
           self.loop = loop
           self.queue: asyncio.Queue[AXEvent] = asyncio.Queue()
           self._thread: Optional[threading.Thread] = None
           self._cfrunloop = None
           self._observers: dict[int, Any] = {}      # pid -> AXObserver
           self._subscriptions: list[Subscription] = []
           self._log = structlog.get_logger()

       def start(self) -> None:
           if self._thread is not None:
               return
           ready = threading.Event()
           def _runloop_target() -> None:
               try:
                   from CoreFoundation import CFRunLoopGetCurrent, CFRunLoopRun
               except ImportError:
                   ready.set()
                   return
               self._cfrunloop = CFRunLoopGetCurrent()
               ready.set()
               CFRunLoopRun()  # blocks until CFRunLoopStop
           self._thread = threading.Thread(target=_runloop_target, name="cua-cfrunloop", daemon=True)
           self._thread.start()
           if not ready.wait(timeout=2.0):
               raise RuntimeError("CFRunLoop thread did not become ready within 2s")

       def stop(self) -> None:
           if self._cfrunloop is None:
               return
           try:
               from CoreFoundation import CFRunLoopStop
               CFRunLoopStop(self._cfrunloop)
           except ImportError:
               pass
           if self._thread:
               self._thread.join(timeout=2.0)
           self._thread = None
           self._cfrunloop = None

       def subscribe(self, pid: int, element: Any, element_key: str, notifications: list[str], action_id: str) -> Subscription:
           """Subscribes BEFORE fire. Records subscription_ts_ns immediately."""
           sub = Subscription(
               pid=pid,
               element_key=element_key,
               notifications=list(notifications),
               action_id=action_id,
               subscription_ts_ns=time.monotonic_ns(),
           )
           # The actual AXObserverCreate + AXObserverAddNotification calls happen on the CFRunLoop thread
           # via dispatch; for Phase 1 we use a synchronous approach because subscribe is called rarely.
           try:
               from HIServices import (
                   AXObserverCreate,
                   AXObserverAddNotification,
                   AXObserverGetRunLoopSource,
               )
               from CoreFoundation import (
                   CFRunLoopAddSource,
                   kCFRunLoopDefaultMode,
               )
           except ImportError:
               # Phase 1 fail-open: subscription remains as a record but no real observer.
               self._subscriptions.append(sub)
               return sub

           def _callback(observer, axelem, notif_name, user_info_ref, refcon):
               event = AXEvent(
                   pid=pid,
                   element_key=element_key,
                   notif=str(notif_name),
                   user_info=user_info_ref,
                   event_ts_ns=time.monotonic_ns(),
                   action_id=str(refcon) if refcon else None,
               )
               # CALLBACK FIRES ON CFRunLoop THREAD — handoff to asyncio thread
               self.loop.call_soon_threadsafe(self.queue.put_nowait, event)

           if pid not in self._observers:
               err, observer = AXObserverCreate(pid, _callback, None)
               if err != 0:
                   raise axerror_from_code(err, f"AXObserverCreate(pid={pid}) failed")
               self._observers[pid] = observer
               source = AXObserverGetRunLoopSource(observer)
               # Add the source to the CFRunLoop thread's run loop
               if self._cfrunloop is not None:
                   CFRunLoopAddSource(self._cfrunloop, source, kCFRunLoopDefaultMode)
               sub._observer = observer
               sub._runloop_source = source

           observer = self._observers[pid]
           for notif in notifications:
               err = AXObserverAddNotification(observer, element, notif, action_id.encode())
               if err != 0 and err != -25208:  # -25208 = kAXErrorNotificationAlreadyRegistered: ok
                   self._log.warning("axobserver.add_notification_failed", code=err, notif=notif)
           self._subscriptions.append(sub)
           return sub
   ```

   ```python
   import asyncio
   from typing import Optional
   import structlog

   from cua_overlay.state.graph import UIElement
   from cua_overlay.ax.observer import AXEventBridge, AXEvent, Subscription

   _GUARD_NS = 5_000_000   # 5 ms; Pitfall P28 mitigation


   class AXObserverManager:
       """Pre-subscribe-before-fire orchestration.

       This is the secret-weapon contract: callers MUST `await expect(...)` BEFORE
       firing the action; otherwise stale notifications can race in.
       """
       def __init__(self, bridge: AXEventBridge) -> None:
           self._bridge = bridge
           self._dispatcher_task: Optional[asyncio.Task] = None
           self._waiters: list[tuple[Subscription, asyncio.Future, set[str]]] = []
           self._log = structlog.get_logger()

       def start(self) -> None:
           """Start the dispatcher task that consumes bridge.queue and routes to waiters."""
           if self._dispatcher_task is None:
               self._dispatcher_task = asyncio.create_task(self._dispatch_loop())

       async def stop(self) -> None:
           if self._dispatcher_task:
               self._dispatcher_task.cancel()
               try:
                   await self._dispatcher_task
               except asyncio.CancelledError:
                   pass
               self._dispatcher_task = None

       async def expect(
           self,
           target: UIElement,
           notifs: list[str],
           action_id: str,
           timeout_ms: int = 500,
           ax_element: Any = None,
       ) -> AXEvent:
           """Subscribe BEFORE the action fires. Returns a future for the first matching event.
           Raises asyncio.TimeoutError on timeout_ms elapsed.
           """
           # ax_element is the raw AXUIElement — caller resolves from target.
           # In Phase 1 demo, the test passes the element directly.
           sub = self._bridge.subscribe(
               pid=target.pid,
               element=ax_element,
               element_key=target.composite_key,
               notifications=notifs,
               action_id=action_id,
           )
           fut: asyncio.Future[AXEvent] = asyncio.get_running_loop().create_future()
           self._waiters.append((sub, fut, set(notifs)))
           try:
               return await asyncio.wait_for(fut, timeout=timeout_ms / 1000.0)
           except asyncio.TimeoutError:
               # Remove waiter on timeout
               self._waiters = [w for w in self._waiters if w[1] is not fut]
               raise

       async def _dispatch_loop(self) -> None:
           while True:
               event = await self._bridge.queue.get()
               # Route to all matching waiters
               matched: list[asyncio.Future] = []
               for sub, fut, notif_set in list(self._waiters):
                   if fut.done():
                       continue
                   if not _passes_filter(event, sub, notif_set):
                       continue
                   if not fut.done():
                       fut.set_result(event)
                       matched.append(fut)
               # Remove matched waiters
               if matched:
                   self._waiters = [w for w in self._waiters if w[1] not in matched]


   def _passes_filter(event: AXEvent, sub: Subscription, notifs: set[str]) -> bool:
       """Three-predicate filter for Pitfall P28 + correctness:

       1. event_ts_ns must be >= subscription_ts_ns + 5_000_000ns (5ms stale guard)
       2. event.action_id must match sub.action_id (don't catch other waiters' events)
       3. event.notif must be in the requested notifs set
       """
       if event.event_ts_ns < sub.subscription_ts_ns + _GUARD_NS:
           return False
       if event.action_id != sub.action_id:
           return False
       if event.notif not in notifs:
           return False
       if event.pid != sub.pid:
           return False
       return True
   ```

   import asyncio
   import os
   import select
   from typing import Callable, Optional
   import structlog

   class KqueueProcObserver:
       """Pure asyncio. Uses loop.add_reader on the kqueue fd.

       EVFILT_PROC + NOTE_EXIT fires when a watched pid exits.
       """
       NOTE_EXIT = 0x80000000
       EVFILT_PROC = -5
       EV_ADD = 0x0001
       EV_DELETE = 0x0002

       def __init__(self, loop: Optional[asyncio.AbstractEventLoop] = None) -> None:
           self.loop = loop or asyncio.get_event_loop()
           self.kq_fd: Optional[int] = None
           self.callbacks: dict[int, Callable[[int], None]] = {}
           self._log = structlog.get_logger()

       async def __aenter__(self) -> "KqueueProcObserver":
           self.start()
           return self

       async def __aexit__(self, *exc) -> None:
           self.stop()

       def start(self) -> None:
           if self.kq_fd is not None:
               return
           self.kq_fd = select.kqueue().fileno()  # NOTE: select.kqueue() returns a kqueue obj — keep ref
           # Actually we need the wrapping kqueue object alive to keep fd valid:
           self._kq = select.kqueue()
           self.kq_fd = self._kq.fileno()
           self.loop.add_reader(self.kq_fd, self._on_readable)

       def stop(self) -> None:
           if self.kq_fd is None:
               return
           self.loop.remove_reader(self.kq_fd)
           self._kq.close()
           self.kq_fd = None
           self.callbacks.clear()

       def watch_pid(self, pid: int, on_exit: Callable[[int], None]) -> None:
           if self.kq_fd is None:
               raise RuntimeError("KqueueProcObserver not started")
           ev = select.kevent(
               ident=pid,
               filter=select.KQ_FILTER_PROC,
               flags=select.KQ_EV_ADD | select.KQ_EV_ENABLE,
               fflags=select.KQ_NOTE_EXIT,
           )
           self._kq.control([ev], 0, 0)
           self.callbacks[pid] = on_exit

       def unwatch_pid(self, pid: int) -> None:
           if self.kq_fd is None:
               return
           ev = select.kevent(
               ident=pid,
               filter=select.KQ_FILTER_PROC,
               flags=select.KQ_EV_DELETE,
           )
           try:
               self._kq.control([ev], 0, 0)
           except Exception:
               pass  # already gone
           self.callbacks.pop(pid, None)

       def _on_readable(self) -> None:
           if self.kq_fd is None:
               return
           events = self._kq.control(None, 16, 0)  # non-blocking read up to 16 events
           for e in events:
               if e.fflags & select.KQ_NOTE_EXIT:
                   pid = int(e.ident)
                   cb = self.callbacks.pop(pid, None)
                   if cb:
                       try:
                           cb(pid)
                       except Exception:
                           self._log.exception("kqueue.callback_error", pid=pid)
   ```

   class NSWorkspaceObserver:
       """Subscribes to NSWorkspaceDidActivateApplicationNotification on a dedicated NSOperationQueue.

       Phase 1 scope: frontmost-app changes only. NSDistributedNotificationCenter integration is
       in distnotif.py (Phase 2 implementation).
       """
       def __init__(self, loop: asyncio.AbstractEventLoop) -> None:
           self.loop = loop
           self._observer = None
           self._callbacks: list[Callable[[str, int], None]] = []   # (bundle_id, pid)
           self._log = structlog.get_logger()

       def start(self) -> None:
           if self._observer is not None:
               return
           try:
               from AppKit import NSWorkspace, NSWorkspaceDidActivateApplicationNotification
               from Foundation import NSOperationQueue
           except ImportError:
               self._log.warning("nsworkspace.pyobjc_unavailable")
               return

           ws = NSWorkspace.sharedWorkspace()
           nc = ws.notificationCenter()
           queue = NSOperationQueue.alloc().init()

           def _handler(notif):
               app = notif.userInfo().get("NSWorkspaceApplicationKey")
               if app is None:
                   return
               bundle_id = str(app.bundleIdentifier() or "")
               pid = int(app.processIdentifier())
               # Hand off to asyncio
               for cb in list(self._callbacks):
                   self.loop.call_soon_threadsafe(cb, bundle_id, pid)

           self._observer = nc.addObserverForName_object_queue_usingBlock_(
               NSWorkspaceDidActivateApplicationNotification,
               None,
               queue,
               _handler,
           )

       def stop(self) -> None:
           if self._observer is None:
               return
           try:
               from AppKit import NSWorkspace
               nc = NSWorkspace.sharedWorkspace().notificationCenter()
               nc.removeObserver_(self._observer)
           except Exception:
               pass
           self._observer = None
           self._callbacks.clear()

       def on_frontmost_change(self, callback: Callable[[str, int], None]) -> None:
           self._callbacks.append(callback)
   ```
