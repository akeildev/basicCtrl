   import json
   import os
   from pathlib import Path
   from typing import Any

   def atomic_write_json(path: Path, data: Any) -> None:
       """Write JSON to <path> atomically via temp file + os.replace.

       Used for snapshot.json and any other crash-resilient single-file writes.
       """
       path.parent.mkdir(parents=True, exist_ok=True)
       tmp = path.with_suffix(path.suffix + ".tmp")
       tmp.write_text(json.dumps(data, indent=2, default=str))
       os.replace(tmp, path)

   def read_json(path: Path) -> Any:
       return json.loads(path.read_text())
   ```

   import json
   import os
   import uuid
   from pathlib import Path
   from typing import Optional, Any

   import structlog

   from basicctrl.persist.snapshot_io import atomic_write_json, read_json


   class SessionWriter:
       """Per-session directory tree under ~/.cua/sessions/<uuid>/.

       Layout per ARCHITECTURE.md L124-130:
       ~/.cua/sessions/<session_id>/
       ├── snapshot.json              # last full StateGraph snapshot (atomic)
       ├── action_log.ndjson          # structlog NDJSON, every Hoare triple
       ├── checkpoints/               # LangGraph checkpoint dumps (Postgres-mirrored)
       ├── recipes/                   # Phase 4
       ├── cassettes/                 # Phase 3
       ├── recordings/                # Phase 5
       ├── heals.ndjson               # Phase 3 heal events (Phase 1: empty)
       └── profile_snapshot/          # AppProfile cache copies for this session
       """
       SUBDIRS: list[str] = ["checkpoints", "recipes", "cassettes", "recordings", "profile_snapshot"]
       EMPTY_FILES: list[str] = ["heals.ndjson"]

       def __init__(self, base: Optional[Path] = None, session_id: Optional[str] = None) -> None:
           self._session_id = session_id or str(uuid.uuid4())
           default_base = Path.home() / ".cua" / "sessions"
           self._base = base or default_base
           self._dir = self._base / self._session_id
           self._dir.mkdir(parents=True, exist_ok=True)
           for sub in self.SUBDIRS:
               (self._dir / sub).mkdir(parents=True, exist_ok=True)
           for f in self.EMPTY_FILES:
               (self._dir / f).touch()
           # Touch action_log so file always exists
           self.action_log_path.touch()
           structlog.get_logger().info("session.created", session_id=self._session_id, dir=str(self._dir))

       @property
       def session_id(self) -> str:
           return self._session_id

       @property
       def dir(self) -> Path:
           return self._dir

       @property
       def action_log_path(self) -> Path:
           return self._dir / "action_log.ndjson"

       @property
       def snapshot_path(self) -> Path:
           return self._dir / "snapshot.json"

       def append_action_log(self, event: dict[str, Any]) -> None:
           """NDJSON append. Each event is one line of valid JSON."""
           with self.action_log_path.open("a", encoding="utf-8") as f:
               f.write(json.dumps(event, default=str))
               f.write("\n")

       def write_snapshot(self, data: Any) -> None:
           atomic_write_json(self.snapshot_path, data)

       def read_snapshot(self) -> Any:
           if not self.snapshot_path.exists():
               return None
           return read_json(self.snapshot_path)
   ```

1. **Create `basicctrl/persist/durable_step.py`**:
   ```python
   from typing import Optional, Any
   import structlog

   from basicctrl.state.causal_dag import HoarePre, HoarePost, ActionCanonical


   _DEFAULT_CONN = "postgresql://localhost:5432/basicctrl"


   class DurableExecutor:
       """Wraps langgraph-checkpoint-postgres AsyncPostgresSaver.

       Per STACK.md: langgraph-checkpoint-postgres 3.0.5; AsyncPostgresSaver.from_conn_string;
       autocommit + dict_row; persists graph state per node.

       Phase 1 scope: minimal — only the verifier loop's checkpoint, not full graph state.
       Phase 6 hardens this for crash-resume under load.
       """
       def __init__(self, conn_string: str = _DEFAULT_CONN) -> None:
           self._conn_string = conn_string
           self._cm = None         # async context manager handle
           self._saver = None      # AsyncPostgresSaver instance
           self._log = structlog.get_logger()

       async def setup(self) -> None:
           """Open the saver and provision tables. Idempotent."""
           if self._saver is not None:
               return
           from langgraph.checkpoint.postgres.aio import AsyncPostgresSaver
           self._cm = AsyncPostgresSaver.from_conn_string(self._conn_string)
           self._saver = await self._cm.__aenter__()
           await self._saver.setup()
           self._log.info("durable.setup_complete", conn=self._mask_conn())

       async def aclose(self) -> None:
           if self._cm is not None:
               try:
                   await self._cm.__aexit__(None, None, None)
               finally:
                   self._cm = None
                   self._saver = None

       async def checkpoint(
           self,
           session_id: str,
           step_idx: int,
           pre: HoarePre,
           action: ActionCanonical,
           post: HoarePost,
       ) -> None:
           """Write a single (pre, action, post) tuple to Postgres."""
           if self._saver is None:
               raise RuntimeError("DurableExecutor not setup() — call setup() before checkpoint()")
           config = {
               "configurable": {
                   "thread_id": session_id,
                   "checkpoint_id": str(step_idx),
               }
           }
           state: dict[str, Any] = {
               "step_idx": step_idx,
               "pre": pre.model_dump(mode="json"),
               "action": action.model_dump(mode="json"),
               "post": post.model_dump(mode="json"),
           }
           # AsyncPostgresSaver.aput signature: (config, checkpoint, metadata, new_versions)
           await self._saver.aput(config, state, metadata={}, new_versions={})
           self._log.info("durable.checkpoint_written", session_id=session_id, step_idx=step_idx)

       async def latest_checkpoint(self, session_id: str) -> Optional[dict]:
           """Return the most recent checkpoint dict for this session_id, or None."""
           if self._saver is None:
               raise RuntimeError("DurableExecutor not setup()")
           config = {"configurable": {"thread_id": session_id}}
           cp = await self._saver.aget(config)
           if cp is None:
               return None
           # cp is the AsyncPostgresSaver checkpoint object — extract our state dict
           if isinstance(cp, dict):
               return cp
           # Some versions return a typed Checkpoint with .channel_values or .values
           values = getattr(cp, "channel_values", None) or getattr(cp, "values", None) or cp
           return values if isinstance(values, dict) else dict(values)

       def _mask_conn(self) -> str:
           # Conservative: don't log the conn string at all if it has anything that looks like a password
           if "@" in self._conn_string and ":" in self._conn_string.split("@")[0]:
               return "postgresql://***@***"
           return self._conn_string
   ```
