   ```python
   import asyncio
   import time
   from dataclasses import dataclass
   from typing import Optional
   import anyio
   import structlog
   from PIL import Image
   import imagehash

   from basicctrl.state.graph import UIElement, Bbox

   _DHASH_THRESHOLD_BITS = 5    # Hamming distance threshold; 5/64 = ~8% change

   @dataclass
   class L1Snapshot:
       window_list: dict[int, dict]
       pasteboard_change_count: int
       roi_dhash: Optional[str]
       captured_at: float


   class L1Cheap:
       """L1 cheap-diff verifier: three sub-checks in parallel.

       Per VERIFY-05 + Pitfall research:
       - L1a CGWindowList diff (~1-2ms): added/removed/title_changed windows
       - L1b NSPasteboard.changeCount (<1ms): integer counter (T-1-03: never log contents)
       - L1c pixel ROI dHash (~10-20ms with screenshot capture): 64-bit fingerprint compare
       """
       def __init__(self) -> None:
           self._log = structlog.get_logger()

       async def snapshot(self, target: UIElement) -> L1Snapshot:
           """Capture pre-action state. Total budget <20ms."""
           t = time.monotonic()
           async with anyio.create_task_group() as tg:
               results: dict = {}
               async def _wl():
                   results["window_list"] = await asyncio.to_thread(self._cgwindowlist_snapshot)
               async def _pb():
                   results["pasteboard"] = await asyncio.to_thread(self._pasteboard_change_count)
               async def _roi():
                   results["roi_dhash"] = await asyncio.to_thread(self._roi_dhash, target.bbox)
               tg.start_soon(_wl)
               tg.start_soon(_pb)
               tg.start_soon(_roi)
           return L1Snapshot(
               window_list=results["window_list"],
               pasteboard_change_count=results["pasteboard"],
               roi_dhash=results["roi_dhash"],
               captured_at=t,
           )

       async def run(self, target: UIElement, before: L1Snapshot) -> dict[str, float]:
           """Capture post-action state in parallel; diff against `before`. Return signal dict."""
           after = await self.snapshot(target)
           signals: dict[str, float] = {}

           # L1a: CGWindowList diff
           wl_diff = self._cgwindowlist_diff(before.window_list, after.window_list)
           changed_count = len(wl_diff["added"]) + len(wl_diff["removed"]) + len(wl_diff["title_changed"])
           signals["l1.window_diff"] = min(1.0, changed_count / 3.0)  # normalize: 3+ changes = 1.0

           # L1b: pasteboard delta (integer-only — T-1-03)
           if after.pasteboard_change_count != before.pasteboard_change_count:
               signals["l1.pasteboard_changed"] = 1.0
               # CRITICAL: do NOT log contents. Only the integer counters.
               self._log.debug("l1.pasteboard_changed",
                              before_count=before.pasteboard_change_count,
                              after_count=after.pasteboard_change_count)
           else:
               signals["l1.pasteboard_changed"] = 0.0

           # L1c: dHash compare
           if before.roi_dhash and after.roi_dhash:
               try:
                   bh = imagehash.hex_to_hash(before.roi_dhash)
                   ah = imagehash.hex_to_hash(after.roi_dhash)
                   distance = abs(bh - ah)  # Hamming distance
                   signals["l1.dhash_changed"] = 1.0 if distance > _DHASH_THRESHOLD_BITS else 0.0
               except Exception as e:
                   self._log.warning("l1.dhash_compare_failed", error=str(e))
                   signals["l1.dhash_changed"] = 0.0
           else:
               signals["l1.dhash_changed"] = 0.0

           return signals

       # ----- helpers (sync, run in to_thread) -----

       def _cgwindowlist_snapshot(self) -> dict[int, dict]:
           try:
               from Quartz import (
                   CGWindowListCopyWindowInfo,
                   kCGWindowListOptionOnScreenOnly,
                   kCGWindowListExcludeDesktopElements,
                   kCGNullWindowID,
               )
               info = CGWindowListCopyWindowInfo(
                   kCGWindowListOptionOnScreenOnly | kCGWindowListExcludeDesktopElements,
                   kCGNullWindowID,
               )
               return {int(w["kCGWindowNumber"]): {
                   "title": str(w.get("kCGWindowName", "")),
                   "owner_pid": int(w.get("kCGWindowOwnerPID", 0)),
                   "level": int(w.get("kCGWindowLayer", 0)),
               } for w in info or []}
           except Exception:
               return {}

       def _cgwindowlist_diff(self, before: dict, after: dict) -> dict:
           return {
               "added": [k for k in after if k not in before],
               "removed": [k for k in before if k not in after],
               "title_changed": [k for k in after if k in before and after[k]["title"] != before[k]["title"]],
           }

       def _pasteboard_change_count(self) -> int:
           """T-1-03: returns integer ONLY. Never returns contents."""
           try:
               from AppKit import NSPasteboard
               return int(NSPasteboard.generalPasteboard().changeCount())
           except Exception:
               return 0

       def _roi_dhash(self, bbox: Bbox) -> Optional[str]:
           """Capture ROI screenshot (50px around bbox.centroid), compute dHash, return hex.

           Phase 1 acceptable to use CGWindowListCreateImage with bounds for the ROI capture.
           Returns None on capture failure (caller treats as "no signal").
           """
           try:
               from Quartz import (
                   CGRectMake,
                   CGWindowListCreateImage,
                   kCGWindowListOptionOnScreenOnly,
                   kCGNullWindowID,
                   kCGWindowImageDefault,
               )
               cx, cy = bbox.centroid
               rect = CGRectMake(cx - 50, cy - 50, 100, 100)
               cg_image = CGWindowListCreateImage(
                   rect,
                   kCGWindowListOptionOnScreenOnly,
                   kCGNullWindowID,
                   kCGWindowImageDefault,
               )
               if cg_image is None:
                   return None
               # Convert CGImage -> PIL via NSBitmapImageRep round-trip
               pil_img = self._cgimage_to_pil(cg_image)
               if pil_img is None:
                   return None
               h = imagehash.dhash(pil_img, hash_size=8)
               return str(h)
           except Exception:
               return None

       def _cgimage_to_pil(self, cg_image) -> Optional[Image.Image]:
           """CGImage → PNG bytes → PIL.Image.
           Phase 1: cheap path via NSBitmapImageRep PNG export."""
           try:
               from AppKit import NSBitmapImageRep, NSPNGFileType
               import io
               rep = NSBitmapImageRep.alloc().initWithCGImage_(cg_image)
               png_data = rep.representationUsingType_properties_(NSPNGFileType, None)
               return Image.open(io.BytesIO(bytes(png_data)))
           except Exception:
               return None
   ```

   ```python
               "l1.window_diff": 0.5,
               "l1.dhash_changed": 0.4,
           },
           "set_value": {
               "ax.value_changed": 0.9,
           },
       }

       def aggregate(self, action_class: str, signals: Mapping[str, float]) -> float:
           # Renormalize by present-signal weights to avoid penalizing
           # missing-but-not-required signals; full weighted-sum tuning
           # happens in Phase 3 against real failure data.
           weights = self.WEIGHTS.get(action_class, {})
           if not weights:
               return 0.0  # unknown action class fails open at 0
           # "active" = signals present in this round with a non-zero value
           active = {sig: w for sig, w in weights.items() if signals.get(sig, 0.0) > 0.0}
           if not active:
               return 0.0
           weighted_sum = sum(signals[sig] * w for sig, w in active.items())
           active_total = sum(active.values())
           result = weighted_sum / active_total
           # Clamp to [0, 1] in case signal values exceed 1.0
           return max(0.0, min(1.0, result))
   ```

   ```python
       L0 is the consumer side: assumes axmgr.expect() was called BEFORE the action fired.

           self,
           axmgr: AXObserverManager,
           ws: Optional[NSWorkspaceObserver] = None,
   ```

   ```python
   import asyncio
   import time
   from typing import Optional
   import anyio
   import structlog

   from basicctrl.state.causal_dag import ActionCanonical, HoarePost
   from basicctrl.state.graph import UIElement
   from basicctrl.verifier.ensemble.l0_push import L0Push
   from basicctrl.verifier.ensemble.l1_cheap import L1Cheap, L1Snapshot
   from basicctrl.verifier.ensemble.weighted_vote import (
       WeightedVote, VERIFIED_THRESHOLD, L3_ESCALATE_THRESHOLD,
   )


   class Aggregator:
       """Top-level verifier entry. Wires L0 push + L1 cheap → WeightedVote → HoarePost.

       Per ARCHITECTURE.md Pattern 4 ("Cheap-deterministic-first ladder"):
       L0 push → L1 cheap (1-5ms) → L2 medium (50-200ms) → L3 LLM (300-800ms).
       Phase 1 wires L0+L1 only. L2 (Plan 06) and L3 (Plan 06 stub) are escalation paths.
       """
       def __init__(self, l0: L0Push, l1: L1Cheap, vote: WeightedVote) -> None:
           self._l0 = l0
           self._l1 = l1
           self._vote = vote
           self._log = structlog.get_logger()

       async def verify(
           self,
           action: ActionCanonical,
           target: UIElement,
           notifs: list[str],
           before_l1: L1Snapshot,
           ax_element: any = None,
           timeout_ms: int = 50,
       ) -> HoarePost:
           """Run L0 + L1 in parallel; aggregate via WeightedVote; return HoarePost.

           Caller MUST have called axmgr.expect() (Plan 04) BEFORE firing the action.
           This verify() is called AFTER firing.
           """
           t_start = time.monotonic()
           l0_signals: dict[str, float] = {}
           l1_signals: dict[str, float] = {}

           async with anyio.create_task_group() as tg:
               async def _l0():
                   nonlocal l0_signals
                   l0_signals = await self._l0.collect(
                       target=target,
                       notifs=notifs,
                       action_id=action.id,
                       timeout_ms=timeout_ms,
                       ax_element=ax_element,
                   )
               async def _l1():
                   nonlocal l1_signals
                   l1_signals = await self._l1.run(target=target, before=before_l1)
               tg.start_soon(_l0)
               tg.start_soon(_l1)

           merged: dict[str, float] = {**l0_signals, **l1_signals}
           confidence = self._vote.aggregate(action.action_type, merged)
           verified = confidence >= VERIFIED_THRESHOLD

           # Per-tier sub-confidence bookkeeping
           tier_signals: dict[str, Optional[float]] = {
               "L0": self._signal_max(l0_signals, prefix="ax."),
               "L1": self._signal_max(l1_signals, prefix="l1."),
               "L2": None,   # Plan 06 fills if escalation needed
               "L3": None,   # Plan 06 stub
           }

           elapsed_ms = (time.monotonic() - t_start) * 1000.0
           self._log.info(
               "verifier.aggregated",
               action_id=action.id,
               action_type=action.action_type,
               target_key=target.composite_key,
               confidence=confidence,
               verified=verified,
               escalate_l3=confidence < L3_ESCALATE_THRESHOLD,
               elapsed_ms=elapsed_ms,
               tier_signals=tier_signals,
           )

           return HoarePost(
               target_key=target.composite_key,
               confidence=confidence,
               tier_signals=tier_signals,
               verified=verified,
               healed_to=None,
               timestamp_ns=time.monotonic_ns(),
           )

       @staticmethod
       def _signal_max(signals: dict[str, float], prefix: str) -> Optional[float]:
           filtered = [v for k, v in signals.items() if k.startswith(prefix)]
           return max(filtered) if filtered else None
   ```
