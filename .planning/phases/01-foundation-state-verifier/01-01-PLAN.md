```python
class Bbox(BaseModel):
    x: float

    INCREMENT = "increment"
    DECREMENT = "decrement"
    SHOWMENU = "show_menu"

class Source(str, Enum):
    AX = "ax"
    CDP = "cdp"

    role: str
    role_path: str
    label: str

    value: Optional[str] = None
```

   ```python
   import structlog
   from structlog.contextvars import merge_contextvars

   _SENSITIVE_FIELDS = {"pasteboard_contents", "clipboard_data", "secrets", "password"}

   def _redact_sensitive(_logger, _method, event_dict):
       """T-1-03 mitigation: strip pasteboard contents, etc.
       Only changeCount integer permitted from L1 pasteboard signal."""
       for k in list(event_dict.keys()):
           if k in _SENSITIVE_FIELDS:
               event_dict[k] = "[REDACTED]"
       return event_dict

   def configure(testing: bool = False) -> None:
       processors = [
           merge_contextvars,
           _redact_sensitive,
           structlog.processors.TimeStamper(fmt="iso", utc=True),
           structlog.processors.add_log_level,
       ]
       if testing:
           processors.append(structlog.processors.dict_tracebacks)
           processors.append(structlog.testing.LogCapture())
       else:
           processors.append(structlog.processors.JSONRenderer())
       structlog.configure(processors=processors, cache_logger_on_first_use=True)
   ```

   ```python
   from collections import deque
   from cua_overlay.state.graph import StateGraph, Edge, EdgeKind

   class StateSnapshot(BaseModel):
       """A frozen view of a StateGraph at a point in time. Used in the ring buffer."""
       nodes: dict[str, UIElement]
       timestamp_ns: int

       @classmethod
       def from_graph(cls, g: StateGraph) -> "StateSnapshot":
           return cls(
               nodes={k: v.model_copy(deep=True) for k, v in g.nodes.items()},
               timestamp_ns=time.monotonic_ns(),
           )

   class TemporalRingBuffer:
       """Last 5 frames, in-memory only. ARCHITECTURE.md L491: "snapshot every 60s, prune nodes >5min stale"."""
       def __init__(self, maxlen: int = 5) -> None:
           self.frames: deque[StateSnapshot] = deque(maxlen=maxlen)
           self.precedes_edges: list[Edge] = []

       def push(self, snapshot: StateSnapshot) -> None:
           self.frames.append(snapshot)
           if len(self.frames) >= 2:
               self._link_precedes(self.frames[-2], self.frames[-1])

       def _link_precedes(self, prev: StateSnapshot, curr: StateSnapshot) -> None:
           for key in curr.nodes:
               if key in prev.nodes:
                   self.precedes_edges.append(Edge(
                       src=key,
                       dst=key,
                       kind=EdgeKind.PRECEDES,
                       timestamp_ns=curr.timestamp_ns,
                   ))
   ```
