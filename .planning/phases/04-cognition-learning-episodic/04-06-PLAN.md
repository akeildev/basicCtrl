   ```python
       async def synthesize(self,

           """
           D-16, D-17: Convert 5min recording (150 typeText + 50 click events) → Recipe JSON.

           - recipe.name = task_label
           - recipe.params: infer from typeText actions (e.g., "email_address", "password")

           - recipe.steps: ActionCanonical sequence with locators
           - recipe.on_failure: per-step recovery hints
           """
   ```

   - Per-step on_failure hints: "if click fails, retry once", "if type fails, clear field and retry"

   ```python
   class EpisodicMemory:
       def __init__(self, faiss_path: str = "~/.cua/episodic.faiss", embedding_model: str = "sentence-transformers/all-MiniLM-L6-v2"):
           self.faiss_path = Path(faiss_path).expanduser()

       async def index_recipe(self, recipe: Recipe, app_bundle_id: str, task_class: str, state_fingerprint: str):
           """
           D-19: Index recipe with (app, task_class, state_fp) key.
           Embed recipe text (name + preconditions + steps serialized).
           Store metadata: success_count=0, failure_count=0, quarantined=False.
           Per D-19: also store embedding_source_text for re-embedding on model swap.
           """
       
       async def lookup(self, 
                       app_bundle_id: str,
                       task_class: str,
                       state_fingerprint: str,
                       top_k: int = 3) -> list[EpisodicHit]:
           """
           D-20: Return top-K recipes with similarity > 0.85.
           Called BEFORE planner LLM (D-20 episodic-first retrieval).
           Quarantined recipes (failure_count > 2) surface with quarantined=True flag.
           """
       
       def mark_recipe_success(self, recipe_id: str):
           """Increment success_count for recipe."""
       
       def mark_recipe_failure(self, recipe_id: str):
           """Increment failure_count; quarantine if > 2."""
   ```
