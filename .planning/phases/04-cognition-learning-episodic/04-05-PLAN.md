1. Create LearningRecorder.swift (~120 LOC, ghost-os pattern):
   ```swift
   import Cocoa
   import os.log
   
   class LearningRecorder {
       private let logger = os.log.OSLog(subsystem: "cua.LearningRecorder", category: "recorder")
       private var eventTap: CGEventTap?
       private let queue = DispatchQueue(label: "com.cua.learning-recorder", qos: .background)
       
       func startRecording() {
           queue.async { [weak self] in
               guard let self = self else { return }
               
               // Create CGEventTap(.listenOnly, not .default)
               let tap = CGEvent.tapCreate(
                   tap: .cghidEventTap,
                   place: .headInsertEventTap,
                   options: .listenOnly,
                   eventsOfInterest: CGEventMask(...),  // keyboard + mouse events
                   callback: self.eventCallback,
                   userInfo: UnsafeMutableRawPointer(Unmanaged.passUnretained(self).toOpaque())
               )
               
               // Register with CFRunLoop on the bg queue
               let runLoop = CFRunLoopGetCurrent()
               CGEventTapEnable(tap, true)
               CFRunLoopAddSource(runLoop, tap, .commonModes)
               self.eventTap = tap
               
               // Run loop
               CFRunLoopRun()
           }
       }
       
       func stopRecording() {
           queue.async { [weak self] in
               if let tap = self?.eventTap {
                   CGEventTapEnable(tap, false)
               }
               CFRunLoopStop(CFRunLoopGetCurrent())
           }
       }
       
       // JSONL over stdout (Phase 1 IPC pattern)
       private func emitEvent(type: String, payload: [String: Any]) {
           let event: [String: Any] = ["type": type, "ts": Date().timeIntervalSince1970, "payload": payload]
           if let json = try? JSONSerialization.data(withJSONObject: event, options: []),
              let jsonStr = String(data: json, encoding: .utf8) {
               print(jsonStr)
           }
       }
   }
   ```

1. Create `basicctrl/learning/recorder.py`:
   ```python
   class LearningRecorder:
       def __init__(self, socket_path: str = "/tmp/cua-learning.sock"):
           self.socket_path = socket_path
           self.coalescer = KeystrokeCoalescer(window_ms=500)  # CFRunLoopTimer 0.5s
       
       async def start(self):
           """Start reading JSONL from Swift tap via unix socket."""
           async with await asyncio.open_unix_connection(path=self.socket_path) as (reader, writer):
               while True:
                   line = await reader.readline()
                   if not line:
                       break
                   event = json.loads(line)
                   self.process_event(event)
       
       def process_event(self, event: dict):
           """Convert CGEvent tap events to ObservedAction."""
           if event["type"] == "keystroke":
               self.coalescer.add_keystroke(event["key"], event["ts"])
           elif event["type"] == "mouse_click":
               coalesced_type_text = self.coalescer.flush()  # Emit buffered keystrokes as 1 typeText
               action = ActionCanonical(
                   action_type="click",
                   user_gesture_type="click",
                   timestamp=event["ts"]
               )
               yield ObservedAction(step_idx=..., action=action, user_gesture_type="click", ...)
   ```

2. Create `basicctrl/learning/coalesce.py`:
   ```python
   class KeystrokeCoalescer:
       def __init__(self, window_ms: float = 500.0):
           self.window_ms = window_ms
           self.buffer: list[tuple[str, float]] = []
           self.last_flush_ts = 0.0
       
       def add_keystroke(self, key: str, ts: float):
           self.buffer.append((key, ts))
           if ts - self.last_flush_ts > self.window_ms:
               return self.flush()
       
       def flush(self) -> str:
           """Convert buffered keystrokes to 1 typeText action per word (whitespace-separated)."""
           if not self.buffer:
               return None
           keys = [k for k, _ in self.buffer]
           text = "".join(keys)
           # Split on whitespace, yield one typeText per word
           words = text.split()
           self.buffer = []
           self.last_flush_ts = max([ts for _, ts in self.buffer]) if self.buffer else time.time()
           return " ".join(words)  # or yield individually if needed
   ```
