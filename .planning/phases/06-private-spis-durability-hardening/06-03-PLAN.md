"""SPI-02: AX remote notifications for occluded-app automation.

Per RESEARCH.md §"AX Remote Notifications (SPI-02)" L78-91:
- Private HIServices SPI: _AXObserverAddNotificationAndCheckRemote
- Enables Slack/Discord/VS Code background automation when occluded
- Already implemented in Phase 2 AXObserver; this formalizes as SPI-02

PITFALL P14 (ROADMAP Phase 1 BLOCKER): AX notifs fail on web/Electron
→ Solution: _AXObserverAddNotificationAndCheckRemote keeps trees alive.
"""
import logging
from typing import Optional

import structlog
from basicctrl.ax.observer import AXObserverManager

log = structlog.get_logger(__name__)


class AXRemoteBridge:
    """Wrapper for AX remote notifications (Phase 2 formalization).
    
    Per ARCHITECTURE.md L8 SPI integration tier:
    Already shipping; this just documents SPI status and gating.
    """
    
    def __init__(self, available: bool = True):
        """
        Args:
            available: Capability probe result (should always be True on Phase 2+)
        """
        self.available = available
        self.observer_manager = AXObserverManager()
        
        if available:
            log.info("ax_remote_bridge_loaded", available=True)
    
    async def subscribe_with_remote_support(self, target_pid, notifications):
        """Subscribe to AX notifications with remote support (keeps occluded trees alive).
        
        Args:
            target_pid: Process ID of Electron app (Slack, Discord, VS Code)
            notifications: List of notification types (kAXValueChanged, etc)
        
        Returns:
            Subscription handle (for later unsubscribe)
        """
        try:
            # Delegate to Phase 2 AXObserver (already has the logic)
            # Phase 2 already uses _AXObserverAddNotificationAndCheckRemote internally
            handle = await self.observer_manager.subscribe(target_pid, notifications)
            
            log.info(
                "ax_remote_subscription_created",
                target_pid=target_pid,
                notification_count=len(notifications),
            )
            return handle
        except Exception as e:
            log.error("ax_remote_subscription_failed", error=str(e))
            raise


# Module-level singleton
_bridge: Optional[AXRemoteBridge] = None


async def get_ax_remote_bridge(capabilities) -> AXRemoteBridge:
    """Get or initialize AX remote bridge.
    
    Args:
        capabilities: SPICapabilities from Wave 0
    
    Returns:
        AXRemoteBridge instance
    """
    global _bridge
    if _bridge is None:
        _bridge = AXRemoteBridge(available=capabilities.ax_remote_available)
    return _bridge
