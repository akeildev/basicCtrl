Per RESEARCH.md §"Per-SPI Status Table" L47:
- Tier B: SIP partial-off required
- Skip gracefully on default Mac
"""
import ctypes
import subprocess
import structlog

log = structlog.get_logger(__name__)

class EndpointSecurityBridge:
    def __init__(self, available: bool = False):
        self.available = available
        if available:
            self._check_sip_status()
    
    def _check_sip_status(self):
        """Check if SIP allows ES (partial-off or full-off)."""
        try:
            result = subprocess.run(
                ['csrutil', 'status'],
                capture_output=True,
                text=True,
                timeout=2
            )
            # If "on", ES has limited access; if off, works fully
            self.available = "on" not in result.stdout.lower()
        except Exception as e:
            log.warning("sip_status_check_failed", error=str(e))
            self.available = False
    
    async def new_client(self):
        """Create ES client (only if SIP allows)."""
        if not self.available:
            log.info("endpoint_security_unavailable_on_default_mac")
            return None
        # Implementation deferred: ES client setup is complex
        return None

async def get_endpoint_security_bridge(capabilities):
    return EndpointSecurityBridge(available=capabilities.endpoint_security_available)

"""SPI-05: DTrace probes (app-internals introspection).

Per RESEARCH.md §"Per-SPI Status Table" L48:
- Tier B: SIP partial-off required
- Skip gracefully on default Mac
"""
import subprocess
import structlog

log = structlog.get_logger(__name__)

class DTraceBridge:
    def __init__(self, available: bool = False):
        self.available = available
        if available:
            log.info("dtrace_bridge_loaded")
    
    async def spawn_probe(self, probe_script: str):
        """Spawn DTrace probe (only if SIP allows)."""
        if not self.available:
            log.info("dtrace_unavailable_on_default_mac")
            return None
        try:
            result = subprocess.run(
                ['dtrace', '-s', '-'],
                input=probe_script,
                capture_output=True,
                timeout=5,
                text=True
            )
            return result.stdout if result.returncode == 0 else None
        except Exception as e:
            log.error("dtrace_probe_failed", error=str(e))
            return None

async def get_dtrace_bridge(capabilities):
    return DTraceBridge(available=capabilities.dtrace_available)
