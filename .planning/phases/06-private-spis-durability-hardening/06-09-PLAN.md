```python
"""Durable execution wrapper via LangGraph PostgresSaver.

Per RESEARCH.md §"Durability Hardening" L243-287:
Each translator call wraps as one LangGraph node.
Checkpoint post-action state (hoare_pre, action, hoare_post).
Resume-from-crash via last verified checkpoint.

Per STACK.md: langgraph-checkpoint-postgres==3.0.5
"""
from datetime import datetime
from langgraph.checkpoint.postgres.aio import AsyncPostgresSaver
from pydantic import BaseModel
import os
import structlog

log = structlog.get_logger(__name__)


class DurableStepCheckpoint(BaseModel):
    """Checkpoint structure per RESEARCH.md L263-270."""
    step_idx: int
    hoare_pre: dict  # Pre-action app state
    action: dict  # ActionCanonical (serialized)
    hoare_post: dict  # Post-action app state
    verifier_status: str  # PENDING | VERIFIED | FAILED
    timestamp: float
    translator_id: str
    channel_id: str


class DurabilityHarness:
    """Wrapper that makes translator calls durable."""
    
    def __init__(self, session_id: str):
        self.session_id = session_id
        self.checkpointer = None
        self._step_idx = 0
        
        # Connection string from env or default
        self.postgres_url = os.getenv(
            "CUA_POSTGRES_URL",
            "postgresql://localhost:5432/cua_maximalist"
        )
    
    async def setup(self):
        """Initialize PostgresSaver (create tables, etc)."""
        self.checkpointer = await AsyncPostgresSaver.from_conn_string(
            self.postgres_url
        ).__aenter__()
        await self.checkpointer.setup()
        log.info("durability_harness_setup", session_id=self.session_id)
    
    async def wrapped_translator_call(self, translator, target, action, hoare_pre):
        """Wrap a translator.fire() call as a durable step.
        
        Args:
            translator: Translator instance (T1-T5)
            target: TranslatorTarget
            action: ActionCanonical
            hoare_pre: HoarePre (pre-action state)
        
        Returns:
            ChannelOutcome (same as translator.fire())
        """
        try:
            # Fire translator
            outcome = await translator.fire(target, action)
            
            # Construct checkpoint
            checkpoint = DurableStepCheckpoint(
                step_idx=self._step_idx,
                hoare_pre=hoare_pre.model_dump(),
                action=action.model_dump(),
                hoare_post={},  # Will be filled by verifier post-action
                verifier_status="PENDING",
                timestamp=datetime.now().timestamp(),
                translator_id=translator.id if hasattr(translator, 'id') else "unknown",
                channel_id=outcome.channel_id,
            )
            
            # Save checkpoint to Postgres
            await self.checkpointer.put(
                checkpoint.model_dump(),
                {"configurable": {"session_id": self.session_id, "step_idx": self._step_idx}},
            )
            
            log.info(
                "durable_step_checkpointed",
                session_id=self.session_id,
                step_idx=self._step_idx,
                verifier_status="PENDING",
            )
            
            self._step_idx += 1
            return outcome
        
        except Exception as e:
            log.error("durable_step_failed", error=str(e), step_idx=self._step_idx)
            raise
    
    async def update_checkpoint_with_verifier(self, step_idx: int, hoare_post: dict, status: str):
        """Update checkpoint after verifier completes.
        
        Args:
            step_idx: Step index to update
            hoare_post: Post-action state
            status: VERIFIED | FAILED
        """
        try:
            checkpoint = await self.checkpointer.get(
                {"configurable": {"session_id": self.session_id, "step_idx": step_idx}},
            )
            if checkpoint:
                checkpoint["hoare_post"] = hoare_post
                checkpoint["verifier_status"] = status
                await self.checkpointer.put(
                    checkpoint,
                    {"configurable": {"session_id": self.session_id, "step_idx": step_idx}},
                )
                log.info(
                    "durable_step_updated",
                    session_id=self.session_id,
                    step_idx=step_idx,
                    verifier_status=status,
                )
        except Exception as e:
            log.error("durable_step_update_failed", error=str(e), step_idx=step_idx)
```

```python
"""Resume from crash via last verified checkpoint.

Per RESEARCH.md §"Resume logic" L276-285:
Load last checkpoint → restore state graph → continue from next step.
"""
from langgraph.checkpoint.postgres.aio import AsyncPostgresSaver
import structlog

log = structlog.get_logger(__name__)


async def resume_from_crash(session_id: str, postgres_url: str):
    """Load last verified checkpoint and resume execution.
    
    Args:
        session_id: Session to resume
        postgres_url: Postgres connection string
    
    Returns:
        (last_verified_step_idx, state_graph) or None if no checkpoint
    """
    try:
        checkpointer = await AsyncPostgresSaver.from_conn_string(postgres_url).__aenter__()
        
        # Query all checkpoints for session, ordered by step_idx desc
        # PostgresSaver doesn't expose a "get latest" method directly,
        # so we fetch via configurable key pattern
        checkpoint = await checkpointer.get(
            {"configurable": {"session_id": session_id}},
        )
        
        if checkpoint and checkpoint.get("verifier_status") == "VERIFIED":
            log.info(
                "resume_from_checkpoint",
                session_id=session_id,
                step_idx=checkpoint.get("step_idx"),
            )
            return checkpoint.get("step_idx"), checkpoint.get("hoare_post")
        else:
            log.warning("no_verified_checkpoint_found", session_id=session_id)
            return None
    
    except Exception as e:
        log.error("resume_checkpoint_load_failed", error=str(e))
        return None
```
