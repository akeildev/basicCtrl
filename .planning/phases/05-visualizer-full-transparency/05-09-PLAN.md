"""Phase 5 test fixtures + pitfall verification."""
import subprocess
from pathlib import Path
import pytest


@pytest.fixture(scope="session", autouse=True)
def verify_pitfalls():
    """Verify pitfall mitigations via grep (runs once per session)."""
    
    swift_visualizer_path = Path("libs/cua-driver/App/Visualizer.swift")
    swift_ghost_cursor_path = Path("libs/cua-driver/App/GhostCursorView.swift")
    swift_highlight_path = Path("libs/cua-driver/App/HighlightOverlayView.swift")
    
    # P9: SCContentFilter usage
    p9_result = subprocess.run(
        ["grep", "-c", "SCContentFilter", str(swift_visualizer_path)],
        capture_output=True
    )
    assert int(p9_result.stdout.decode() or "0") >= 1, "P9: SCContentFilter missing"
    
    # P10: No sharingType=.none
    p10_result = subprocess.run(
        ["grep", "-c", "sharingType.*\\.none", str(swift_visualizer_path)],
        capture_output=True
    )
    assert int(p10_result.stdout.decode() or "0") == 0, "P10: sharingType=.none detected"
    
    # P11: CAShapeLayer present
    p11_result = subprocess.run(
        ["grep", "-c", "CAShapeLayer", str(swift_highlight_path)],
        capture_output=True
    )
    assert int(p11_result.stdout.decode() or "0") >= 1, "P11: CAShapeLayer missing"
    
    # P12: NSView.draw override present, no CALayer animation
    p12_draw = subprocess.run(
        ["grep", "-c", "override func draw", str(swift_ghost_cursor_path)],
        capture_output=True
    )
    p12_anim = subprocess.run(
        ["grep", "-c", "CABasicAnimation\\|CAKeyframeAnimation", str(swift_ghost_cursor_path)],
        capture_output=True
    )
    assert int(p12_draw.stdout.decode() or "0") >= 1, "P12: NSView.draw missing"
    assert int(p12_anim.stdout.decode() or "0") == 0, "P12: CALayer animation detected"
    
    yield  # Allow tests to run
