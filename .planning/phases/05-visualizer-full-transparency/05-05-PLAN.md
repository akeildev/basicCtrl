```python
"""Visualizer driver — sends commands to Swift sidecar over unix socket."""
import asyncio
import json
from pathlib import Path

from basicctrl.visualizer.models import (
    GhostCursorCommand, HighlightBoxCommand, HUDCommand
)


class VisualizerBus:
    """Unix socket IPC to Visualizer.swift."""
    
    SOCKET_PATH = Path("/tmp/cua-visualizer.sock")
    
    @staticmethod
    async def send_command(cmd: dict) -> None:
        """Send NDJSON command to visualizer."""
        try:
            reader, writer = await asyncio.open_unix_connection(str(VisualizerBus.SOCKET_PATH))
            writer.write((json.dumps(cmd) + "\n").encode())
            await writer.drain()
            writer.close()
            await writer.wait_closed()
        except (FileNotFoundError, ConnectionRefusedError):
            pass  # Socket not ready
    
    @staticmethod
    async def send_ghost_cursor(x: float, y: float, duration_ms: int) -> None:
        """Animate ghost cursor to target."""
        import time
        cmd = {
            "cmd": "ghost_cursor",
            "x": x,
            "y": y,
            "duration_ms": duration_ms,
            "timestamp_ns": int(time.time_ns()),
        }
        await VisualizerBus.send_command(cmd)
    
    @staticmethod
    async def send_highlight(x: float, y: float, width: float, height: float, label: str, tier: str, channel: str) -> None:
        """Show element highlight box."""
        import time
        cmd = {
            "cmd": "highlight",
            "bbox_x": x,
            "bbox_y": y,
            "bbox_width": width,
            "bbox_height": height,
            "label": label[:40],
            "tier": tier,
            "channel": channel,
            "timestamp_ns": int(time.time_ns()),
        }
        await VisualizerBus.send_command(cmd)
```
