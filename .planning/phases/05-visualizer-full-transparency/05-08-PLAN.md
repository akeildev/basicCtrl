```python
"""Session differ — LCS alignment of action_log.ndjson from two sessions."""
import json
from pathlib import Path
from typing import Optional, List, Tuple


def lcs_alignment(seq_a: List[dict], seq_b: List[dict]) -> List[Tuple[Optional[int], Optional[int]]]:
    """Longest common subsequence alignment.
    
    Returns list of (index_a, index_b) pairs.
    - (i, j): matched steps
    - (i, None): removed from A
    - (None, j): added in B
    """
    # Simplified LCS on (app, target_label, action_type) tuple
    def match_key(action):
        return (action.get("app"), action.get("target_label"), action.get("action_type"))
    
    keys_a = [match_key(a) for a in seq_a]
    keys_b = [match_key(b) for b in seq_b]
    
    # DP table
    m, n = len(keys_a), len(keys_b)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if keys_a[i - 1] == keys_b[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
    
    # Backtrack
    alignment = []
    i, j = m, n
    while i > 0 or j > 0:
        if i > 0 and j > 0 and keys_a[i - 1] == keys_b[j - 1]:
            alignment.append((i - 1, j - 1))
            i -= 1
            j -= 1
        elif j > 0 and (i == 0 or dp[i][j - 1] >= dp[i - 1][j]):
            alignment.append((None, j - 1))
            j -= 1
        else:
            alignment.append((i - 1, None))
            i -= 1
    
    return list(reversed(alignment))


class SessionDiffer:
    """Compare two session action_logs."""
    
    def __init__(self, session_a_id: str, session_b_id: str):
        self.session_a = self._load_session(session_a_id)
        self.session_b = self._load_session(session_b_id)
        self.heals_a = self._load_heals(session_a_id)
        self.heals_b = self._load_heals(session_b_id)
    
    def _load_session(self, session_id: str) -> List[dict]:
        path = Path.home() / ".cua" / "sessions" / session_id / "action_log.ndjson"
        actions = []
        if path.exists():
            with open(path) as f:
                for line in f:
                    if line.strip():
                        actions.append(json.loads(line))
        return actions
    
    def _load_heals(self, session_id: str) -> List[dict]:
        path = Path.home() / ".cua" / "sessions" / session_id / "heals.ndjson"
        heals = []
        if path.exists():
            with open(path) as f:
                for line in f:
                    if line.strip():
                        heals.append(json.loads(line))
        return heals
    
    def generate_diff(self) -> List[dict]:
        """Generate diff output with alignment markers."""
        alignment = lcs_alignment(self.session_a, self.session_b)
        
        diff = []
        for idx_a, idx_b in alignment:
            if idx_a is not None and idx_b is not None:
                # Matched — check for translator swap (heal)
                action_a = self.session_a[idx_a]
                action_b = self.session_b[idx_b]
                
                if action_a.get("tier") != action_b.get("tier"):
                    diff.append({
                        "marker": "healed",
                        "action_a": action_a,
                        "action_b": action_b,
                        "heal_reason": f"{action_a.get('tier')}→{action_b.get('tier')}",
                    })
                else:
                    diff.append({
                        "marker": "same",
                        "action_a": action_a,
                        "action_b": action_b,
                    })
            elif idx_a is not None:
                diff.append({"marker": "removed", "action_a": self.session_a[idx_a]})
            else:
                diff.append({"marker": "added", "action_b": self.session_b[idx_b]})
        
        return diff
```

Pattern: LCS alignment. O(N²) but N typically <100 steps. Returns aligned pairs with markers.

```swift
import SwiftUI

struct SessionDiffView: View {
    let diffItems: [DiffItem]
    @State var showOnlyDiffs = false
    
    var body: some View {
        HStack(spacing: 0) {
            // Session A (left)
            VStack(alignment: .leading) {
                ForEach(diffItems) { item in
                    if !showOnlyDiffs || item.marker != "same" {
                        HStack {
                            if let action = item.action_a {
                                ActionRow(action: action)
                            } else {
                                Text("(removed)")
                                    .font(.system(size: 12))
                                    .foregroundColor(.gray)
                            }
                            Spacer()
                        }
                    }
                }
            }
            .frame(maxWidth: .infinity)
            .padding(.horizontal, 8)
            
            // Diff markers (center)
            VStack {
                ForEach(diffItems) { item in
                    Text(item.marker)
                        .font(.system(size: 10, weight: .medium))
                        .foregroundColor(item.markerColor)
                }
            }
            .frame(width: 60)
            
            // Session B (right)
            VStack(alignment: .leading) {
                ForEach(diffItems) { item in
                    if !showOnlyDiffs || item.marker != "same" {
                        HStack {
                            if let action = item.action_b {
                                ActionRow(action: action)
                            } else {
                                Text("(added)")
                                    .font(.system(size: 12))
                                    .foregroundColor(.gray)
                            }
                            Spacer()
                        }
                    }
                }
            }
            .frame(maxWidth: .infinity)
            .padding(.horizontal, 8)
        }
    }
}

struct DiffItem: Identifiable {
    let id: UUID = UUID()
    let marker: String  // "same", "healed", "added", "removed"
    let action_a: [String: Any]?
    let action_b: [String: Any]?
    
    var markerColor: Color {
        switch marker {
        case "same": return .gray
        case "healed": return .orange
        case "added": return .green
        case "removed": return .red
        default: return .gray
        }
    }
}

struct ActionRow: View {
    let action: [String: Any]
    
    var body: some View {
        HStack(spacing: 4) {
            if let tier = action["tier"] as? String {
                Text(tier)
                    .font(.system(size: 11, weight: .semibold, design: .monospaced))
                    .foregroundColor(.blue)
            }
            
            if let type = action["action_type"] as? String {
                Text(type)
                    .font(.system(size: 11, weight: .regular))
            }
        }
    }
}
```

Pattern: Side-by-side SwiftUI layout. Center column shows markers (same/healed/added/removed).
