```swift
import ScreenCaptureKit
import AVFoundation
import VideoToolbox

/// H.265 screen recorder — 60fps, <16ms latency target, SCContentFilter excludes overlay.
class ScreenRecorder: NSObject {
    private var stream: SCStream?
    private var assetWriter: AVAssetWriter?
    private var writerInput: AVAssetWriterInput?
    private var adaptor: AVAssetWriterInputPixelBufferAdaptor?
    private var encoder: VTCompressionSession?
    private var frameCount = 0
    private let recordingPath: String
    private let metadataPath: String
    private let overlayWindowID: CGWindowID
    
    init(sessionID: String, overlayWindowID: CGWindowID) {
        self.overlayWindowID = overlayWindowID
        let sessionDir = NSSearchPathForDirectoriesInDomains(.applicationSupportDirectory, .userDomainMask, true)[0] + "/.cua/sessions/\(sessionID)"
        self.recordingPath = sessionDir + "/recording.mov"
        self.metadataPath = sessionDir + "/recording_metadata.ndjson"
        super.init()
    }
    
    func startRecording() async throws {
        // Get main display
        guard let display = await SCShareableContent.excludingDesktopWindows().displays.first else {
            throw NSError(domain: "ScreenRecorder", code: 1, userInfo: [NSLocalizedDescriptionKey: "No display found"])
        }
        
        // Setup SCContentFilter to EXCLUDE overlay window (P9/P10)
        let contentFilter = SCContentFilter(display: display, excludingWindows: [overlayWindowID])
        
        // Configure stream
        let streamConfig = SCStreamConfiguration()
        streamConfig.sourceResolution = display.frame.size
        streamConfig.width = Int(display.frame.size.width)
        streamConfig.height = Int(display.frame.size.height)
        streamConfig.frameRate = 60  // 60fps per UI-SPEC
        
        // Create stream
        stream = try await SCStream(filter: contentFilter, configuration: streamConfig, delegate: self)
        
        // Setup H.265 encoder
        setupH265Encoder(width: Int(display.frame.size.width), height: Int(display.frame.size.height))
        
        // Setup AVAssetWriter
        try setupAssetWriter()
        
        // Start capture
        try await stream?.startCapture()
    }
    
    func stopRecording() async throws {
        try await stream?.stopCapture()
        finalize()
    }
    
    private func setupH265Encoder(width: Int, height: Int) {
        var encoder: VTCompressionSession?
        let status = VTCompressionSessionCreate(
            allocator: nil,
            width: Int32(width),
            height: Int32(height),
            codecType: kCMVideoCodecType_HEVC,
            encoderSpecification: nil,
            imageBufferAttributes: nil,
            compressedDataAllocator: nil,
            outputCallback: { refcon, frame, status, infoFlags, sampleBuffer in
                // Handle encoded frame
                if status == noErr, let sampleBuffer = sampleBuffer {
                    if let data = CMSampleBufferGetDataBuffer(sampleBuffer) {
                        CMBlockBufferGetDataPointer(data, atOffset: 0, lengthAtOffsetOut: nil, totalLengthOut: nil, dataPointerOut: nil)
                    }
                }
            },
            refcon: nil,
            compressionSessionOut: &encoder
        )
        
        guard status == noErr, let enc = encoder else { return }
        
        // Optimize for RealTime (P9 — low latency)
        VTSessionSetProperty(enc, key: kVTCompressionPropertyKey_RealTime, value: kCFBooleanTrue)
        VTSessionSetProperty(enc, key: kVTCompressionPropertyKey_TargetQualityForStreaming, value: 100 as CFNumber)
        VTSessionSetProperty(enc, key: kVTCompressionPropertyKey_MaxKeyFrameInterval, value: 30 as CFNumber)  // 30 GOP
        
        self.encoder = enc
    }
    
    private func setupAssetWriter() throws {
        let url = URL(fileURLWithPath: recordingPath)
        assetWriter = try AVAssetWriter(outputURL: url, fileType: .mov)
        
        guard let writer = assetWriter else { return }
        
        // H.265 output settings
        let settings: [String: Any] = [
            AVVideoCodecKey: AVVideoCodecType.hevc,
            AVVideoWidthKey: 1440,  // or detect from display
            AVVideoHeightKey: 900,
        ]
        
        writerInput = AVAssetWriterInput(mediaType: .video, outputSettings: settings)
        
        if let input = writerInput, writer.canAdd(input) {
            writer.add(input)
        }
        
        adaptor = AVAssetWriterInputPixelBufferAdaptor(assetWriterInput: writerInput!)
        try writer.startWriting()
        writer.startSessionAtSourceTime(CMTime.zero)
    }
    
    private func finalize() {
        assetWriter?.finishWriting()
        encoder.flatMap { VTCompressionSessionInvalidate($0) }
    }
}

extension ScreenRecorder: SCStreamDelegate {
    nonisolated func stream(_ stream: SCStream, didOutputSampleBuffer sampleBuffer: CMSampleBuffer, of kind: SCStreamOutputType) {
        // Encode and write frame
        frameCount += 1
    }
}
```

```python
"""Recorder driver — starts/stops ScreenRecorder.swift via IPC."""
import asyncio
import json
from datetime import datetime, timezone
from pathlib import Path

from cua_overlay.observability.session_storage import SessionWriter
from cua_overlay.visualizer.models import ReplayFrameMetadata


class RecorderDriver:
    """Async recorder control."""
    
    def __init__(self, session_writer: SessionWriter):
        self.writer = session_writer
        self.recording = False
        self.frame_idx = 0
    
    async def start(self, overlay_window_id: int) -> None:
        """Start recording via ScreenRecorder.swift."""
        self.recording = True
        self.frame_idx = 0
        # TODO: IPC to Swift ScreenRecorder (Wave 3 integration)
    
    async def stop(self) -> None:
        """Stop recording."""
        self.recording = False
    
    async def write_frame_metadata(self, step_idx: int = None) -> None:
        """Write frame metadata to recording_metadata.ndjson."""
        if not self.recording:
            return
        
        frame = ReplayFrameMetadata(
            frame_idx=self.frame_idx,
            step_idx=step_idx,
            timestamp_ms=int(datetime.now(timezone.utc).timestamp() * 1000),
        )
        self.writer.write_recording_metadata(frame)
        self.frame_idx += 1
```
